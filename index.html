# Ingestion & Reasoning Engine from langgraph.graph import StateGraph from langchain_community import PGVector from aws_bedrock import BedrockLLM # Define Agentic routing state class AgentState(TypedDict): query: str context: list # Load vector index and set up graph db = PGVector(collection="embeddings") llm = BedrockLLM(model_id="claude-3.5-sonnet") workflow = StateGraph(AgentState) workflow.add_node("retrieve", db.similarity_search) workflow.add_node("reason", llm.invoke) workflow.set_entry_point("retrieve") workflow.add_edge("retrieve", "reason") rag_agent = workflow.compile() print("[✓] Agent loaded.
